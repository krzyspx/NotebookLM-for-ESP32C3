#define led_czerw 7  //
#define boot 9       //przycisk boot
#define sww 1
#define led_wew 8

#include <Adafruit_NeoPixel.h>
#define LED_PIN 7  // 10 dla ZERO 7 dla C3 MINI i C3 PICO
#define LED_COUNT 1
#define BRIGHTNESS 50  // Set BRIGHTNESS to about 1/5 (max = 255)

int R = 255;
int G = 0;
int B = 0;

Adafruit_NeoPixel strip(LED_COUNT, LED_PIN, NEO_GRBW + NEO_KHZ800);


void ledsetup() {
  pinMode(led_czerw, OUTPUT);  //LED na module
  digitalWrite(led_czerw, LOW);

  pinMode(led_wew, OUTPUT);  //LED na module
  digitalWrite(led_wew, LOW);

  pinMode(boot, INPUT_PULLUP);
  pinMode(sww, INPUT_PULLUP);

  strip.begin();  // INITIALIZE NeoPixel strip object (REQUIRED)
  strip.show();   // Turn OFF all pixels ASAP
  strip.setBrightness(BRIGHTNESS);
}

int stanswOnOff = 1;
void swOnOff() {

}
int bootOnOff = 1;
void testboot() {

}

}
