//=============================================== setup

void setup() {
  ledsetup();
  if (procesor == D1MINI) Serial.begin(115200);

  delay(1000);
  Serial.println();
  Serial.println(F(__FILE__));
  // Serial.setDebugOutput(true);

esp_task_wdt_init(WDT_TIMEOUT, true); //włączenie obsługi watchdog ESP32
esp_task_wdt_add(NULL); //dodanie bieżącego wątku do "pilnowania" przez WDT

  akcja.attach(0, 3001, led_app_timer);  // miganie led w aplikacji
  akcja.attach(1, 10000, I2C_timer);     // odczyt czujników I2C
  akcja.attach(2, 2003, led_timer);       // miganie led w ESP
 // akcja.attach(3, 10000, program_timer);  // timer dla programu - czas pompy
  akcja.attach(4, 2500, tft_timer);       // timer dla programu - czas pompy
 // akcja.attach(5, 5 * 60 * 1001, sensorT_timer);  //


  connectToWiFiNetwork();
  otasetup();
  webSocket.begin();
  webSocket.onEvent(webSocketEvent);
  webSocket.enableHeartbeat(5000, 3000, 2);

  ntpsetup();    //setup for NTP clock
  no433setup();  //setup 433 MHz
  //dallasset(); //setup for ds18b20
  programsetup();  //setup for my program
  tftsetup();
  I2Csetup();
}

//===================================================== loop

void loop() {
  akcja.process();  // timer
  ArduinoOTA.handle();
  webSocket.loop();

  no433akcja();    //433 Mhz
  sw_boot_test();  // sprawdzanie stanu przełącznikow
  //  czy_programowanie() // programownie pilotów
}
